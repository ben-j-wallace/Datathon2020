Model

model <- lm(unemployment_change ~ retail_avg + 
              grocery_avg + 
              workplaces_avg + 
              residential_avg, 
            data = merged_data)
int_only_model <- lm(unemployment_change ~ 1, data = merged_data)

final_model <- step(model, scope = formula(int_only_model), direction = "backward")
